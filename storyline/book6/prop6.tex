\begin{minipage}[t]{0.54\textwidth}
    \begin{center}
        \textit{PROPOSITION VI. THEOREM.}\phantomsection\label{book6pr6} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=3, loversize=1, nindent=0pt]{\euclidinitials{I}}{}F \textit{two triangles} (\hspace{-1ex}$\img[0][0][30]{right_triangle_11}$ \textit{and}\\ $\img[0][0][30]{top_triangle_3}$\hspace{-1ex}) \textit{have one angle}
    \end{center}
    \raggedright (\hspace{-1ex}$\img{right_red_angle_7}$\hspace{-1ex}) \textit{of the one, equal to one angle}\\ (\hspace{-1ex}$\img{red_arc_angle_2}$\hspace{-1ex}) \textit{of the other, and the ſides about the equal angles proportional, the triangles ſhall be equiangular, and have thoſe angles equal which the homologous ſides ſubtend}.
\end{minipage}

\hfill

\hfill

\begin{center}
    From the extremities of $\blackline \bcomma$ one of the ſides\\
    of $\img[0][0][30]{top_triangle_3} \bcomma$ about $\img{red_arc_angle_2} \bcomma$ draw\\
    $\yellowline$ and $\dottedyellowline \bcomma$ making\\
    $\img{left_red_angle_7} \bequals \img{right_red_angle_7} \bcomma$ and $\img{left_blue_angle_8} \bequals \img{right_blue_angle_7} \bsemicolon$ then $\img{black_angle_47} \bequals \img{yellow_angle_58}$ [\hyperref[book1pr32]{\textsc{I.} 32}], and the two triangles being equiangular,\\
    $\dottedblueline \bcolon \dottedblackline \bbcolon \dottedyellowline \bcolon \blackline$ [\hyperref[book6pr4]{\textsc{VI.} 4}];\\
    but $\dottedblueline \bcolon \dottedblackline \bbcolon \blueline \bcolon \blackline$ [hyp.];\\
    $\btherefore \dottedyellowline \bcolon \blackline \bbcolon \blueline \bcolon \blackline$ [\hyperref[book5pr11]{\textsc{V.} 11}],\\
    and conſequently $\dottedyellowline \bequals \blueline$ [\hyperref[book5pr9]{\textsc{V.} 9}];\\
    $\btherefore \img[0][0][30]{top_triangle_3} \bequals \img[0][0][30]{bottom_triangle_4}$  in every reſpect [\hyperref[book1pr4]{\textsc{I.} 4}].
\end{center}

\begin{center}
    But $\img{left_blue_angle_8} \bequals \img{right_blue_angle_7}$ [conſt.],\\
    and $\btherefore \img{blue_arc_angle} \bequals \img{right_blue_angle_7} \bsemicolon$ and\\
    ſince also $\img{red_arc_angle_2} \bequals \img{right_red_angle_7} \bcomma$\\
    $\img{yellow_arc_angle} \bequals \img{yellow_angle_58}$ [\hyperref[book1pr32]{\textsc{I.} 32}];\\
    and $\btherefore \img[0][0][30]{right_triangle_11}$ and $\img[0][0][30]{top_triangle_3}$ are equiangular, with their equal angles oppoſite to homologous ſides.
\end{center}
