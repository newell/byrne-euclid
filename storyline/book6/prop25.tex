        \textit{PROPOSITION XXV. PROBLEM.}\phantomsection\label{book6pr25} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=4, loversize=1, nindent=0pt]{\euclidinitials{T}}{}O \textit{deſcribe a rectilinear figure which ſhall be\\ ſimilar to a given rectilinear figure} (\hspace{-1ex}$\img{red_triangle_15}$\hspace{-1ex}), \textit{and\\ equal to another} (\hspace{-1ex}$\img{polygon}$\hspace{-1ex}].
    \end{center}

    \hfill

    \hfill

    \begin{center}
        Upon $\yellowline$ deſcribe $\img{blue_rectangle_4} \bequals \img{red_triangle_15} \bcomma$\\
        and upon $\redline$ deſcribe $\img{outlined_rectangle_and_yellow_angle} \bequals \img{polygon} \bcomma$\\
        and having $\img[0][0][20]{yellow_angle_29} \bequals \img[0][0][20]{red_angle_54}$  [\hyperref[book1pr45]{\textsc{I.} 45}], and then\\
        $\yellowline$ and $\dottedblackline$ will lie in the ſame ſtraight line [\textsc{I.} \hyperref[book1pr29]{29}, \hyperref[book1pr14]{14}].\\
        \hfill\\
        Between $\yellowline$ and $\dottedblackline$ find a mean proportional\\
        $\blueline$ [\hyperref[book6pr13]{\textsc{VI.} 13}], and upon $\blueline$\\
        deſcribe $\img{yellow_triangle_10} \bcomma$ ſimilar to $\img{red_triangle_15} \bcomma$ and ſimilarly ſituated.\\
        \hfill\\
        Then $\img{yellow_triangle_10} \bequals \img{polygon} \bperiod$
    \end{center}

    \hfill

    \begin{center}
        For ſince $\img{red_triangle_15}$ and $\img{yellow_triangle_10}$ are ſimilar, and\\
        $\yellowline \bcolon \blueline \bbcolon \blueline \bcolon \dottedblackline$ [conſt.],\\
        $\img{red_triangle_15} \bcolon \img{yellow_triangle_10} \bbcolon \yellowline \bcolon \dottedblackline$ [\hyperref[book6pr20]{\textsc{VI.} 20}];\\
        but $\img{blue_rectangle_4} \bcolon \img{outlined_rectangle_and_yellow_angle} \bbcolon \yellowline \bcolon \dottedblackline$ [\hyperref[book6pr1]{\textsc{VI.} 1}];\\
        $\btherefore \img{red_triangle_15} \bcolon \img{yellow_triangle_10} \bbcolon \img{blue_rectangle_4} \bcolon \img{outlined_rectangle_and_yellow_angle}$ [\hyperref[book5pr11]{\textsc{V.} 11}];
    \end{center}
