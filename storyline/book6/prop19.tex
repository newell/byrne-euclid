\begin{minipage}[t]{0.54\textwidth}
    \begin{center}
        \textit{PROPOSITION XIX. THEOREM.}\phantomsection\label{book6pr19} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=3, loversize=1, nindent=0pt]{\euclidinitials{S}}{}IMILAR \textit{triangles} (\hspace{-1ex}$\img[0][0][20]{yellow_triangle_7}$ \textit{and} $\img{red_and_blue_triangle}$\hspace{-1ex}) \textit{are to one another}
    \end{center}
    \raggedright \textit{in the duplicate ratio of their homologous ſides}.
\end{minipage}

\hfill

\hfill

\raggedright Let $\img[0][0][20]{black_angle_41}$ and $\img[0][0][20]{red_angle_52}$ be equal angles, and $\bimg{dotted_black_and_black_line_2}$ and let $\blueline$ homologous ſides of the ſimilar triangles $\img[0][0][20]{yellow_triangle_7}$ and $\img{red_and_blue_triangle}$ and on $\bimg{dotted_black_and_black_line_2}$ the greater of theſe lines take $\dottedblackline$ a third proportional, ſo that

\begin{center}
    $\bimg{dotted_black_and_black_line_2} \bcolon \blueline \bbcolon \blueline \bcolon \dottedblackline \bsemicolon$\\
    draw $\dottedyellowline \bperiod$\\
    \hfill\\
    $\bimg{dotted_black_and_black_line_2} \bcolon \yellowline \bbcolon \blueline \bcolon \redline$ [\hyperref[book6pr4]{\textsc{VI.} 4}];\\
    $\btherefore \blackline \bcolon \blueline \bbcolon \yellowline \bcolon \redline$ [\hyperref[book5pr16]{\textsc{V.} 16}, \hyperref[book5def13]{alt}],\\
    but $\bimg{dotted_black_and_black_line_2} \bcolon \blueline \bbcolon \blueline \bcolon \dottedblackline$ [conſt.],\\
    $\btherefore \blueline \bcolon \dottedblackline \bbcolon \yellowline \bcolon \redline$
\end{center}

\begin{center}
    conſequently $\img[0][0][20]{yellow_triangle_7} \bequals \img[0][0][17]{blue_triangle_10}$ for they have the ſides about\\
    the equal angles $\img[0][0][20]{black_angle_41}$ and $\img[0][0][20]{red_angle_52}$ reciprocally proportional \mbox{[\hyperref[book6pr15]{\textsc{VI.} 15}]};\\
\end{center}

\begin{center}
    $\btherefore \img{red_and_blue_triangle} \bcolon \img[0][0][20]{yellow_triangle_7} \bbcolon \img{red_and_blue_triangle} \bcolon \img[0][0][17]{blue_triangle_10}$ [\hyperref[book5pr7]{\textsc{V.} 7}];\\
    but $\img{red_and_blue_triangle} \bcolon \img[0][0][17]{blue_triangle_10} \bbcolon \bimg{dotted_black_and_black_line_2} \bcolon \dottedblackline$ [\hyperref[book6pr1]{\textsc{VI.} 1}],\\
    $\btherefore \img{red_and_blue_triangle} \bcolon \img[0][0][20]{yellow_triangle_7} \bbcolon \bimg{dotted_black_and_black_line_2} \bcolon \dottedblackline \bcomma$\\
    that is to ſay, the triangles are to one another in the duplicate ratio of their homologous ſides\\
    $\blueline$ and $\bimg{dotted_black_and_black_line_2}$ [\hyperref[book5def11]{\textsc{V.} def. 11}].
\end{center}

\hfill

\hfill Q.E.D.
