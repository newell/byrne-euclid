\begin{minipage}[t]{0.64\textwidth}
    \vspace{0pt}

    \begin{center}
        \textit{PROPOSITION XVII. THEOREM.}\phantomsection\label{book6pr17} \\
    \end{center}

    \hfill

    \begin{center}
        PART I.\\
        \raggedright \lettrine[lines=4, loversize=1, nindent=0pt]{\euclidinitials{I}}{}F \textit{three ſtraight lines be proportional}\\ (\hspace{-1ex}$\redline \bcolon \blueline \bbcolon \blueline \bcolon \blackline$\hspace{-1ex}) \textit{the\\ rectangle under the extremes is equal to the ſquare\\ of the mean}.
    \end{center}

    \hfill

    \begin{center}
        PART II.\\
        \vspace{1ex}
        \raggedright \textit{And if the rectangle under the extremes be equal to the ſquare of the mean, the three ſtraight lines are proportional}.
    \end{center}

    \hfill

    \hfill

    \begin{center}
        PART I.\\
        \vspace{1ex}
        Aſſume $\yellowline \bequals \blueline \bcomma$ and\\
        ſince $\redline \bcolon \blueline \bbcolon \blueline \bcolon \blackline \bcomma$\\
        then $\redline \bcolon \blueline \bbcolon \yellowline \bcolon \blackline \bcomma$\\
        $\btherefore \redline \bcross \blackline \bequals \blueline \bcross \yellowline$ [\hyperref[book6pr16]{\textsc{VI.} 16}].
    \end{center}

    \hfill

    \begin{center}
        But $\yellowline \bequals \blueline \bcomma$\\
        $\btherefore \blueline \bcross \yellowline \bequals \blueline \bcross \blueline \bcomma$ or $\bequals \blueline^{\text{\large 2}} \bsemicolon$\\
        therefore, if the three ſtraight lines are proportional, the rectangle contained by the extremes is equal to the ſquare of the mean.
    \end{center}
\end{minipage}

\hfill

\hfill

\pagebreak

\begin{minipage}[t]{0.64\textwidth}
    \vspace{0pt}

    \begin{center}
        PART II.\\
        \vspace{1ex}
        Aſſume $\yellowline \bequals \blueline \bcomma$ then\\
        $\redline \bcross \blackline \bequals \yellowline \bcross \blueline \bperiod$\\
        $\btherefore \redline \bcolon \blueline \bbcolon \yellowline \bcolon \blackline$ [\hyperref[book6pr16]{\textsc{VI.} 16}], and \\
        $\btherefore \redline \bcolon \blueline \bbcolon \blueline \bcolon \blackline \bperiod$
    \end{center}

    \hfill

    \hfill Q.E.D.
