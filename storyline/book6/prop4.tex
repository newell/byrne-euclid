\begin{minipage}[t]{0.54\textwidth}
    \begin{center}
        \textit{PROPOSITION IV. THEOREM.}\phantomsection\label{book6pr4} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=3, loversize=1, nindent=0pt]{\euclidinitials{I}}{}N \textit{equiangular triangles}\\ (\hspace{-1ex}$\img[0][0][20]{left_triangle_11}$ \textit{and} $\img[0][0][30]{right_triangle_10}$\hspace{-1ex}) \textit{the ſides}
    \end{center}
    \vspace{2ex}
    \raggedright \textit{about the equal angles are proportional, and the ſides which are oppoſite to the equal angles are homologous}.
\end{minipage}

\hfill

\raggedright Let the equiangular triangles be ſo placed that two ſides $\blackline \bcomma$\\
$\dottedblackline$ oppoſite to equal angles $\img{red_arc_angle}$ and $\img{black_arc_angle}$ may be conterminous and in the ſame ſtraight line; and that the triangles lying at the ſame ſide of that ſtraight line, may have the equal angles not conterminous,

\begin{center}
    i.e. $\img{red_angle_49}$ oppoſite to $\img{yellow_angle_57} \bcomma$ and $\img{blue_angle_52}$ to $\img{black_angle_39} \bperiod$
\end{center}

\begin{center}
    Draw $\dottedyellowline$ and $\yellowline \bperiod$ Then, becauſe\\
    $\img{blue_angle_52} \bequals \img{black_angle_39} \bcomma \redline \bparallel \bimg{yellow_and_dotted_red_line}$ [\hyperref[book1pr28]{\textsc{I.} 28}];\\
    and for a like reaſon, $\dottedblueline \bparallel \bimg{blue_and_dotted_yellow_line} \bcomma$\\
    $\btherefore \img[0][0][45]{parallelogram_3}$ is a parallelogram.\\
    But $\blackline \bcolon \dottedblackline \bbcolon \yellowline \bcolon \dottedredline$ [\hyperref[book6pr2]{\textsc{VI.} 2}];\\
    and ſince $\yellowline \bequals \redline$ [\hyperref[book1pr34]{\textsc{I.} 34}],\\
    $\blackline \bcolon \dottedblackline \bbcolon \redline \bcolon \dottedredline \bsemicolon$ and by\\
    alternation, $\blackline \bcolon \redline \bbcolon \dottedblackline \bcolon \dottedredline$ [\hyperref[book5pr16]{\textsc{V.} 16}].
\end{center}

\begin{center}
    In like manner it may be ſhown, that\\
    $\blueline \bcolon \dottedblueline \bbcolon \blackline \bcolon \dottedblackline \bsemicolon$\\
    and by alternation, that\\
    $\blueline \bcolon \blackline \bbcolon \dottedblueline \bcolon \dottedblackline \bsemicolon$\\
    but it has already been proved that\\
    $\blackline \bcolon \redline \bbcolon \dottedblackline \bcolon \dottedredline \bcomma$\\
    and therefore, ex \ae quali,\\
    $\blueline \bcolon \redline \bbcolon \dottedblueline \bcolon \dottedredline \bcomma$ [\hyperref[book5pr22]{\textsc{V.} 22}],\\
    therefore the ſides about the equal angles are proportional, and thoſe which are oppoſite to the equal angles are homologous.
\end{center}

\hfill
