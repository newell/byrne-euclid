\begin{minipage}[t]{0.54\textwidth}
    \begin{center}
        \textit{PROPOSITION X. PROBLEM.}\phantomsection\label{book4pr10} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=3, loversize=1, nindent=0pt]{\euclidinitials{T}}{}O \textit{conſtruct an iſoſceles triangle, in which each of the angles at the\\ baſe ſhall be double of the}
    \end{center}
    \textit{vertical angle}.
\end{minipage}%

\hfill

\begin{center}
    Take any ſtraight line $\bimg{black_and_dotted_black_line_12}$ and divide it ſo that\\
    $\bimg{black_and_dotted_black_line_12} \bcross \dottedblackline \bequals \blackline^{\text{\large 2}}$ [\hyperref[book2pr11]{\textsc{II.} 11}].
\end{center}

\begin{center}
    With $\bimg{black_and_dotted_black_line_12}$ as radius, deſcribe $\img[0][0][30]{red_circle_2}$ and place\\
    in it from the extremity of the radius, $\blueline \bequals \blackline \bcomma$ \mbox{[\hyperref[book4pr1]{\textsc{IV.} 1}]};\\
    draw $\yellowline \bperiod$
\end{center}

\begin{center}
    Then $\img[0][0][30]{large_triangle_2}$ is the required triangle.\\
    For, draw $\redline$ and deſcribe\\
    $\img[0][0][20]{blue_circle_3}$ about $\img[0][0][30]{left_triangle_7}$ [\hyperref[book4pr5]{\textsc{IV.} 5}].\\
    Since $\bimg{black_and_dotted_black_line_12} \bcross \dottedblackline \bequals \blackline^{\text{\large 2}} \bequals \blueline^{\text{\large 2}} \bcomma$\\
    $\btherefore \blueline$ is tangent to $\img[0][0][20]{blue_circle_3}$ [\hyperref[book3pr37]{\textsc{III.} 37}].\\
    $\btherefore \img[0][0][15]{yellow_angle_52} \bequals \img{outlined_angle_4}$ [\hyperref[book3pr32]{\textsc{III.} 32}],\\
    add $\img[0][0][20]{black_angle_34}$ to each,\\
    $\btherefore \img[0][0][15]{yellow_angle_52} \bplus \img[0][0][20]{black_angle_34} \bequals \img{outlined_angle_4} \bplus \img[0][0][20]{black_angle_34} \bsemicolon$\\
    but $\img[0][0][20]{black_angle_34} \bplus \img[0][0][15]{yellow_angle_52}$ or $\img{black_and_yellow_angle} \bequals \img{red_angle_44}$ [\hyperref[book1pr5]{\textsc{I.} 5}]:
    ſince $\yellowline \bequals \bimg{black_and_dotted_black_line_12}$ [\hyperref[book1pr5]{\textsc{I.} 5}]\\
    conſequently $\img{red_angle_44} \bequals \img{outlined_angle_4} \bplus \img[0][0][20]{black_angle_34} \bequals \img{blue_angle_47}$ [\hyperref[book1pr32]{\textsc{I.} 32}]
\end{center}

\begin{center}
    $\btherefore \redline \bequals \blueline$ [\hyperref[book2pr6]{\textsc{II.} 6}]\\
    $\btherefore \blueline \bequals \blackline \bequals \redline$ [conſt.]\\
    $\btherefore \img{outlined_angle_4} \bequals \img[0][0][20]{black_angle_34}$ [\hyperref[book1pr5]{\textsc{I.} 5}]
\end{center}

\begin{center}
    $\btherefore \img{red_angle_44} \bequals \img{black_and_yellow_angle} \bequals \img{blue_angle_47} \bequals \img{outlined_angle_4} \bplus \img[0][0][20]{black_angle_34} \bequals$ twice $\img{outlined_angle_4} \bsemicolon$\\
    and conſequently each angle at the baſe is double of the vertical angle.
\end{center}

\hfill
