    \begin{center}
        \textit{PROPOSITION XVI. THEOREM.}\phantomsection\label{book3pr16} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=3, loversize=1, nindent=0pt]{\euclidinitials{T}}{}HE \textit{ſtraight line} $\yellowline$ \textit{drawn from the extremity of the diameter} $\blackline$ \textit{of a circle perpendicular to it falls without the circle. And if any ſtraight line} $\dottedredline$ \textit{be}
    \end{center}
\end{minipage}
\textit{drawn from a point within that perpendicular to the point of contact, it cuts the circle}.

\begin{center}
    PART I.\\
    \hfill\\
    If it be poſſible, let $\redline \bcomma$ which meets the circle again, be $\bperp \blackline \bcomma$ and draw $\blueline \bperiod$
\end{center}

\begin{center}
    Then, becauſe $\blueline \bequals \blackline \bcomma \img[0][0][20]{yellow_angle_35} \bequals \img{black_angle_21}$ [\hyperref[book1pr5]{\textsc{I.} 5}],\\
    and $\btherefore$ each of these angles is acute [\hyperref[book1pr17]{\textsc{I.} 17}]\\
    but $\img[0][0][20]{yellow_angle_35} \bequals \img{right_angle_2}$ [hyp.], which is abſurd, therefore $\redline$ drawn $\bperp \blackline$ does not meet the circle again.
\end{center}

\hfill

\begin{center}
    PART II.\\
    \hfill\\
    \raggedright Let $\yellowline$ be $\bperp \blackline$ and let $\dottedredline$ be drawn from a point
    $\img{point_red_and_black_dotted_lines}$ between $\yellowline$ and the circle, which if it be poſſible, does not cut the circle.
\end{center}

\hfill

\begin{center}
    Becauſe $\img{blue_and_red_angle} \bequals \img{right_angle_2} \bcomma$\\
    $\btherefore \img{blue_angle_30}$ is an acute angle; ſuppoſe\\
    $\bimg{dotted_blue_and_black_line} \bperp \dottedredline \bcomma$ drawn from the centre of the circle, it muſt fall at the ſide of $\img{blue_angle_30}$ the acute angle.\\
    $\btherefore \img{outlined_angle}$ which is ſuppoſed to be a right angle, is $\bgt \img{blue_angle_30} \bcomma$\\
    $\btherefore \blackline \bgt \bimg{dotted_blue_and_black_line} \bsemicolon$\\
    but $\dottedblueline \bequals \blackline \bcomma$
\end{center}

\hfill

\raggedright and $\btherefore \dottedblueline \bgt \bimg{dotted_blue_and_black_line} \bcomma$  a part greater than the whole, which is abſurd. Therefore the point does not fall outside the circle, and therefore the ſtraight line $\dottedredline$ cuts the circle.

\hfill

\hfill Q.E.D.
