    \vspace{0pt}

    \begin{center}
        \textit{PROPOSITION XIX. THEOREM.}\phantomsection\label{book1pr19} \\
    \end{center}

    \hfill

    \begin{center}
        \raggedright \lettrine[lines=4, loversize=1, nindent=0pt]{\euclidinitials{I}}{}F \textit{in any triangle} $\img[0][0][30]{triangle_8}$ \textit{one angle} $\img{blue_angle_12}$ \textit{be\\ greater than another} $\img{red_angle_7}$ \textit{the ſide} $\blueline$\\ \textit{which is oppoſite to the greater angle, is greater}
    \end{center}
    \textit{than the ſide} $\redline$ \textit{oppoſite the leſs}.

    \hfill

    \hfill

    \begin{center}
        If $\blueline$ be not greater than $\redline$ then  muſt\\
        $\blueline \bequals$ or $\blt \redline \bperiod$\\
        \hfill \\
        If $\blueline \bequals \redline$ then\\
        $\img{blue_angle_12} \blt \img{red_angle_7}$ [\hyperref[book1pr5]{\textsc{I.} 5}];\\
        which is contrary to the hypotheſis.\\
        $\blueline$ is not leſs than $\redline \bsemicolon$ for if it were,\\
        $\img{blue_angle_12} \blt \img{red_angle_7}$ [\hyperref[book1pr18]{\textsc{I.} 18}]\\
        which is contrary to the hypotheſis:\\
        $\btherefore \blueline \bgt \redline \bperiod$
    \end{center}

    \hfill

    \hfill Q.E.D.
\end{minipage}%
